# Gnrer une squence d'tats alatoire de taille n pour le modele SIR
import numpy as np
import random as rd

def generer_etats_aleatoire(n):
    """ en entre: n : la taille de la squence
        en sortie:  liste_etat : liste representant la sequence d'etat
                    seq : une chaine de charactere representant la sequence
                    proba : la probabilit de la squence
                    k+1 : la taille de la squence de sortie
                    p0 : la distribution de probabilit
    """
    #choix de l'tat initial au hasard
    seq="" # la squence d'etat
    liste_etat=[]
    matriceProba1=np.array([[0.9,0.1,0.0],[0.0,0.5,0.5],[0.0,0.0,1.0]])
    
    s=rd.randint(0,2)
    if(s==0):
        p0=[0.9,0.1,0.0]
        seq+="S "
        liste_etat.append('S')
    elif(s==1):
        p0=[0.0,0.5,0.5]
        seq+="I "
        liste_etat.append('I')
    elif(s==2):
        p0=[0.0,0.0,1]
        seq+="R "
        liste_etat.append('R')
    else:
        print("error")
    
    proba=1/3 # P(X0) probabilit qu'on soit dans un des tats (S, I, R) au dbut
    matriceProba2=matriceProba1
    p1=p0
    i=s
    k=1
    for k in range(1,n):
        j=rd.randint(0,2)
        while(matriceProba1[i,j]==0):#pour eviter de gnerer une squence avec une probabilit nulle
            j=rd.randint(0,2)
        if(j==0):
            seq+="S "
            liste_etat.append('S')
        elif(j==1):
            seq+="I "
            liste_etat.append('I')
        elif(j==2):
            seq+="R "
            liste_etat.append('R')
        else:
            print("error")
        
        proba*=matriceProba1[i,j]  # proba de la squence
        p0=np.dot(p1,matriceProba2) # la distribution de probabilit
        matriceProba2=np.dot(matriceProba2,matriceProba1)
        i=j
            
    
    return liste_etat, seq, proba, k+1, p0

def generer_etats_init_S(n):
    """ gnre une sequence d'tat  partir de l'tat S (Sain)
        en entre: n : la taille de la squence
        en sortie:  liste_etat : liste representant la sequence d'etat
                    seq : une chaine de charactere representant la sequence
                    proba : la probabilit de la squence
                    k+1 : la taille de la squence de sortie
                    p0 : la distribution de probabilit
    """
    #choix de l'tat initial au hasard
    seq="" # la squence d'etat
    liste_etat=[]
    matriceProba1=np.array([[0.0,1.0,0.0],[0.0,0.5,0.5],[0.0,0.0,1.0]])
    
    s=0
    if(s==0):
        p0=[0.0,1.0,0.0]
        seq+="S "
        liste_etat.append('S')
   
    
    proba=1/3 # P(X0) probabilit qu'on soit dans un des tats (S, I, R) au dbut
    matriceProba2=matriceProba1
    p1=p0
    i=s
    k=1
    for k in range(1,n):
        j=rd.randint(0,2)
        while(matriceProba1[i,j]==0):#pour eviter de gnerer une squence avec une probabilit nulle
            j=rd.randint(0,2)
        if(j==0):
            seq+="S "
            liste_etat.append('S')
        elif(j==1):
            seq+="I "
            liste_etat.append('I')
        elif(j==2):
            seq+="R "
            liste_etat.append('R')
        else:
            print("error")
        
        proba*=matriceProba1[i,j]  # proba de la squence
        p0=np.dot(p1,matriceProba2) # la distribution de probabilit
        matriceProba2=np.dot(matriceProba2,matriceProba1)
        i=j
            
    
    return liste_etat, seq, proba, k+1, p0

5.5 Critiques du modèle SIR et Améliorations :¶

-Entre l'tat Sain 'S' et l'tat Infect 'I', pour certaines maladies, lorsque un individu sain est y expos il faut parfois un certain temps avant que l'individu ne soit infect, ce qui est appele priode de latence, du coup en devrait rajouter un autre tat Expos 'E' entre l'tat S et I. -Aprs qu'un individu a t infect, il peut dcder, reprsent par l'tat Dcs 'D'. La maladie peut se terminer d'elle mme et donner l'individu une immunisation contre la rinfection, reprsent par l'tat Guri 'R'. Dans le cas contraire, l'individu peut toujours tre infectieux, il se retrouve isol de la population par politique de quarantaine, reprsent par l'tat Quarantaine 'Q', sans ngliger le fait que ce dernier peut ventuellement gurir et rintgrer l'tat Sain 'S'.
